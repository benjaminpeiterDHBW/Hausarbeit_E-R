\newcommand{\mytable}[5][tab]{%
  \begin{table}[H]
    \centering
    \footnotesize
    \setlength\extrarowheight{2pt}
    \begin{tabular}{#2}
      #5
    \end{tabular}
    \vspace{-2mm}
    \caption{#3}
    \label{#1:#4}
  \end{table}   }

%% Table - Appendix
% \mytable{Columns}{Caption}{Label}{Table}
\newcounter{appendixcounter}
\newcounter{oldtable}
\newcommand{\mytableappendix}[4]{%
  \renewcommand{\tablename}{Appendix}
  \setcounter{oldtable}{\value{table}}
  \setcounter{table}{\value{appendixcounter}}
  \mytable[app]{#1}{#2}{#3}{#4}
  \stepcounter{appendixcounter}
  \setcounter{table}{\value{oldtable}}   }

%% Figure - Appendix
% \myfigure{figure.eps}{Caption}{Label}{Width}
\newcounter{oldfigure}
\newcommand{\myfigureappendix}[4]{%
  \renewcommand{\figurename}{Appendix}
  \setcounter{oldfigure}{\value{figure}}
  \setcounter{figure}{\value{appendixcounter}}
  \myfigure[app]{#1}{#2}{#3}{#4}
  \stepcounter{appendixcounter}
  \setcounter{figure}{\value{oldfigure}}  }
